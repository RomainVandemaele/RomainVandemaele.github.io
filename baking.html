    class  LemonCrinkles {

        HashMap ingredients = new  HashMap ();
    
        public LemonCrinkles(Pantry pantry) {
    
            //unit in gr or ml for liquids
            ingredients.put("flour", 300);
            ingredients.put("butter", 100);
            ingredients.put("egg", 1);
            ingredients.put("lemon juice", 150);
            ingredients.put("baking soda", 5);
            ingredients.put("granulated sugar", 300);
            ingredients.put("caster sugar", 15);
    
            fetchIngredients(Pantry);
            Oven = new  Oven();
            oven.preheat(180);
    
            Preparation[]  preps = Prepmaker();
            oven.put(preps);
            wait(20*60); //20 min
            this  = oven.pop(); 
            oven.turnOff();
    
            this.eat(); //munch mumch
        }
    
    
        public  void  fetchIngredients(Pantry pantry) {
    
            for (Entry ingredient : ingredients.entrySet()) {
                try  {
                    pantry.fetch(ingredient.getKey(), ingredient.getValue());
                }catch  ( IngredientNotFoundException e ) {
                    throw new  RuntimeException("Go to supermarket");
                }catch  (IngredientQuantityInsufficientException e) {
                    throw new  RuntimeException("Go to supermarket");
                }
            }
        }
    
        public Preparation[] crinklesMaker() {
            Preparation prep = new Preparation();
    
            prep.add( ingredients.get("butter") );
            prep.add( ingredients.get("egg") );
            prep.whisk();
    
            prep.add( ingredients.get("flour") );
            prep.add( ingredients.get("baking soda") );
            prep.add( ingredients.get("granulated sugar") );
            prep.whisk();
    
            prep.add( ingredients.get("lemon") );
            prep.whisk();
    
            Fridge fridge = new  preps Frigde();
            Fridge.put(prep);
            wait(30*60);
            prep = fridge.pop();
    
            Preparation[] preps = prep.divide(20); //divide in 20 smaller ball
            preps.forAll( prep.add( ingredients.get("caster sugar") ) );
            fridge.put(preps);
            wait(30*60);
            preps = fridge.pop();
    
            return  preps;
    
        }
